## 模拟命令行

这是一个命令行模拟的例子：

<div class="terminal">
    <span class="prompt">[~]$</span> <span id="command">whoami</span>
    <div id="output" class="output"></div>
</div>

<script>
    // 模拟命令行输入和输出
    function showOutput() {
        const commandElement = document.getElementById('command');
        const outputElement = document.getElementById('output');

        // 显示命令输入
        commandElement.innerText = 'whoami';

        // 延迟显示命令输出
        setTimeout(() => {
            outputElement.innerText = '好戏在后头. Read more..';
        }, 1000); // 延时1秒后显示输出
    }

    window.onload = showOutput;  // 页面加载时执行
</script>
